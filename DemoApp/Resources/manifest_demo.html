📄 manifest.json 示例

{ "version": "1.0", "app": { "name": "My App", "theme": "auto" }, "tabBar": { "items": [ { "id": "home", "page": "home.html", "title": "首页", "icon": "https://cdn.com/home.png", "iconSelected": "https://cdn.com/home-active.png" } ], "style": { "backgroundColor": "#ffffff", "selectedColor": "#667eea", "unselectedColor": "#999999" } }, "preload": ["home.html", "profile.html"] }

💻 Swift 使用示例

// 1. 设置 manifest URL TabConfigurationManager.shared.setManifestURL( URL(string: "https://cdn.example.com/manifest.json") ) // 2. 加载配置 TabConfigurationManager.shared.loadConfiguration { result in switch result { case .success(let config): print("App: \(config.app.name)") print("Tabs: \(config.tabBar.items.count)") case .failure(let error): print("Error: \(error)") } } // 3. 创建 Tab Bar Controller TabConfigurationManager.shared.createTabBarController { result in switch result { case .success(let tabBarController): window?.rootViewController = tabBarController case .failure(let error): print("Error: \(error)") } } // 4. 预加载页面（应用启动时） TabConfigurationManager.shared.preloadPages { success, total in print("Preloaded \(success)/\(total) pages") }
